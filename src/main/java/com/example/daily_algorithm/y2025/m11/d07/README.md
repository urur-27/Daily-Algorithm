public class Reverse_String {
    public void reverseString(char[] s) {
        int left = 0;
        int right = s.length - 1;
        char extra;

        while (left < right){
            extra = s[right];
            s[right] = s[left];
            s[left] = extra;
            left++;
            right--;
        }
    }
}
